#include <tunables/global>

profile replicarr flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Python runtime
  /usr/bin/python3* mr,
  /usr/lib/python3*/ r,
  /usr/lib/python3*/** mr,
  /usr/local/lib/python3*/ r,
  /usr/local/lib/python3*/** mr,

  # App files
  /app/ r,
  /app/** mr,

  # Persistent data volume
  /data/ r,
  /data/** rw,

  # Capabilities required by s6-overlay-suexec (privilege drop)
  capability setuid,
  capability setgid,
  capability dac_override,
  capability dac_read_search,
  capability chown,
  capability fowner,

  # s6-overlay v3 init and commands
  /init rix,
  /command/ r,
  /command/** rix,
  /package/ r,
  /package/** rix,
  /etc/s6-overlay/ r,
  /etc/s6-overlay/** r,

  # /run — parent dir needs rw so s6 can mkdir /run/s6 and /run/service
  /run/ rw,
  /run/** rw,
  /var/run/ rw,
  /var/run/** rw,

  # proc — s6 reads /proc/self during suexec
  @{PROC}/self/** r,
  @{PROC}/@{pid}/** r,

  # bashio / contenv
  /usr/bin/with-contenv rix,
  /usr/bin/bashio rix,
  /usr/bin/s6* rix,
  /usr/lib/bashio/ r,
  /usr/lib/bashio/** r,

  # Shell and basic utilities needed by run scripts
  /bin/bash rix,
  /bin/sh rix,
  /usr/bin/env rix,
  /usr/bin/** rix,
  /bin/** rix,
  /sbin/** rix,
  /usr/sbin/** rix,

  # Temp files (bashio writes config here)
  /tmp/ rw,
  /tmp/** rw,

  # Network (outbound HTTP to Syncthing instances)
  network tcp,
  network udp,

  # Deny everything else
  deny /proc/sysrq-trigger rw,
  deny /sys/firmware/** rw,
}
